rmvn.Choleski <-
function(n, mu, Sigma) {
# generate n random vectors from MVN(mu, Sigma)
# dimension is inferred from mu and Sigma
d <- length(mu) 
Q <- chol(Sigma) # Choleski factorization of Sigma
Z <- matrix(rnorm(n*d), nrow=n, ncol=d)
X <- Z %*% Q + matrix(mu, n, d, byrow=TRUE)
X
}

Rizzo 5.14

Obtain a Monte Carlo estimate of \[\int_{1}^{\infty} \frac{x^2}{\sqrt 2 \pi} e^{−x^{2}/2} dx\] by importance sampling.

g = function (x) {
  x ^ 2 / sqrt(2*pi) * exp(-x^2/2) * (x > 1)
}

sigma.rayleigh = 1.5
mean = 1.5
n = 10000

f1 = function (x) {
  drayleigh(x) * (x > 1)
}

f2 = function (x) {
  dnorm(x, mean = mean) * (x > 1)
}

rf1 = function () {
  rrayleigh(n)
}

rf2 = function () {
  rnorm(n, mean = mean)
}

is.rayleigh = function () {
  xs = rf1()
  return(mean(g(xs)/f1(xs), na.rm = TRUE))  
}

is.norm = function () {
  xs = rf2()
  return(mean(g(xs)/f2(xs), na.rm = TRUE))  
}

(theta1 = is.rayleigh())
[1] 0.6610559
(theta2 = is.norm())
[1] 0.5777139
(truth = 0.400626)
